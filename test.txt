<build>
  <plugins>
    <plugin>
      <groupId>io.swagger.codegen.v3</groupId>
      <artifactId>swagger-codegen-maven-plugin</artifactId>
      <version>3.0.27</version> <!-- check for the latest version -->
      <executions>
        <execution>
          <goals>
            <goal>generate</goal>
          </goals>
          <configuration>
            <inputSpec>${project.basedir}/src/main/resources/api.yaml</inputSpec> <!-- path to your Swagger YAML -->
            <language>java</language> <!-- target language -->
            <output>${project.build.directory}/generated-sources</output> <!-- output directory -->
            <apiPackage>com.mycompany.app.api</apiPackage> <!-- package for generated API classes -->
            <modelPackage>com.mycompany.app.model</modelPackage> <!-- package for generated model classes -->
            <generateApiTests>false</generateApiTests> <!-- whether to generate API tests -->
            <generateModelTests>false</generateModelTests> <!-- whether to generate model tests -->
          </configuration>
        </execution>
      </executions>
    </plugin>
  </plugins>
</build>
